bank_freqs = [(None, 200), (200, 400), (400, 800), (800, 1600), (1600, 3200), (3200, None)]

def make_filter(freqs):
    if not freqs[0]:
        btype = 'lowpass'
        freqs = freqs[1] / (sr / 2)
    elif not freqs[1]:
        btype = 'highpass'
        freqs = freqs[0] / (sr / 2)
    else:
        btype = 'bandpass'
        freqs = freqs[0] / (sr / 2), freqs[1] / (sr / 2)
    return iirfilter(3, freqs, rp=3, rs=40, btype=btype, ftype='ellip')

def make_fir_filter(freqs):
    if not freqs[0]:
        pass_zero = True
        freqs = freqs[1]
    elif not freqs[1]:
        pass_zero = False
        freqs = freqs[0]
    else:
        pass_zero = False
    return firwin(127, freqs, pass_zero=pass_zero, nyq=sr/2), 1

with Pool() as pool:
    filterbank = pool.map(make_fir_filter, bank_freqs)

we expect the bpm to be the lowest frequency with lots of overtones that has peaks in the spectrum. for now, try every bpm in a big range, and find the one with the strongest fundemental plus overtones
