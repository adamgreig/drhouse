bank_freqs = [(None, 200), (200, 400), (400, 800), (800, 1600), (1600, 3200), (3200, None)]

def make_filter(freqs):
    if not freqs[0]:
        btype = 'lowpass'
        freqs = freqs[1] / (sr / 2)
    elif not freqs[1]:
        btype = 'highpass'
        freqs = freqs[0] / (sr / 2)
    else:
        btype = 'bandpass'
        freqs = freqs[0] / (sr / 2), freqs[1] / (sr / 2)
    return iirfilter(3, freqs, rp=3, rs=40, btype=btype, ftype='ellip')

def make_fir_filter(freqs):
    if not freqs[0]:
        pass_zero = True
        freqs = freqs[1]
    elif not freqs[1]:
        pass_zero = False
        freqs = freqs[0]
    else:
        pass_zero = False
    return firwin(127, freqs, pass_zero=pass_zero, nyq=sr/2), 1

with Pool() as pool:
    filterbank = pool.map(make_fir_filter, bank_freqs)

def half_wave_rectify(x):
    return x.clip(0)

def downsample(x):

n = envelopes_ds[0].size * 2
d = 1/dssr

def freq_to_index(freq):
    return int(freq * ((n * d) / 2))

we expect the bpm to be the lowest frequency with lots of overtones that has peaks in the spectrum. for now, try every bpm in a big range, and find the one with the strongest fundemental plus overtones

In [490]: 
max_evidence = 0
best_bpm = None
evidences = []
bpms = np.arange(50, 210, 0.1)
for bpm in bpms:
    bps = bpm / 60.0
    tones = np.array([freq_to_index(bps * i) for i in (1, 2, 4, 8)])
    evidence = np.sum(fftsum[tones])
    evidence += np.sum(fftsum[tones - 1])
    evidence += np.sum(fftsum[tones + 1])
    if evidence > max_evidence:
        max_evidence = evidence
        best_bpm = bpm
    evidences.append(evidence)
print(best_bpm)
plt.figure(figsize=(18, 5))
plt.plot(bpms, evidences)
plt.grid()
